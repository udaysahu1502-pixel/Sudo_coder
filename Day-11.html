Student CRUD API

A complete Express.js backend for student management

API Documentation

GET /api/students

Retrieve all students

GET /api/students/:id

Retrieve a specific student by ID

POST /api/students

Create a new student

PUT /api/students/:id

Update a student by ID

DELETE /api/students/:id

Delete a student by ID

Example Student Object

{
  "id": 1,
  "name": "John Doe",
  "email": "john.doe@example.com",
  "age": 20,
  "major": "Computer Science",
  "year": "Sophomore"
}

Server Setup Code

const express = require('express');
const app = express();
const PORT = 3000;

app.use(express.json());

let students = [
  { id: 1, name: 'John Doe', email: 'john@example.com', age: 20, major: 'Computer Science', year: 'Sophomore' },
  { id: 2, name: 'Jane Smith', email: 'jane@example.com', age: 21, major: 'Mathematics', year: 'Junior' }
];

// GET all students
app.get('/api/students', (req, res) => {
  res.json(students);
});

// GET student by ID
app.get('/api/students/:id', (req, res) => {
  const student = students.find(s => s.id === parseInt(req.params.id));
  if (!student) return res.status(404).json({ error: 'Student not found' });
  res.json(student);
});

// POST create new student
app.post('/api/students', (req, res) => {
  const { name, email, age, major, year } = req.body;
  const newStudent = {
    id: students.length + 1,
    name,
    email,
    age,
    major,
    year
  };
  students.push(newStudent);
  res.status(201).json(newStudent);
});

// PUT update student
app.put('/api/students/:id', (req, res) => {
  const student = students.find(s => s.id === parseInt(req.params.id));
  if (!student) return res.status(404).json({ error: 'Student not found' });
  
  const { name, email, age, major, year } = req.body;
  student.name = name || student.name;
  student.email = email || student.email;
  student.age = age || student.age;
  student.major = major || student.major;
  student.year = year || student.year;
  
  res.json(student);
});

// DELETE student
app.delete('/api/students/:id', (req, res) => {
  const index = students.findIndex(s => s.id === parseInt(req.params.id));
  if (index === -1) return res.status(404).json({ error: 'Student not found' });
  
  students.splice(index, 1);
  res.status(204).send();
});

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
